corredor:
	1-id -> int(primaria)
	2-corredor -> varchar
	3-id_equipo -> int(indice)
	4-edad -> int
	5-especialidad -> varchar

equipo:
	1-id_equipo -> int(primaria)
	2-equipo -> varchar
	3-division -> varchar
	


SELECT corredor.corredor, corredor.especialidad, equipo.equipo
FROM equipo
INNER JOIN corredor ON corredor.id_equipo=equipo.id_equipo
